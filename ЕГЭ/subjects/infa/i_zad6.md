# Памятка:
нужно запомнить импорт библиотеки, а именно конструкцию
``` python
from turtle import *
```

## а так же знать функции черепахи:
```python
# отключает анимацию черепахи
tracer(0)

# изменяет ширину и длинну поля 
screensize()

#передвижение черепахи: 
forward()
backward()
lt()
rt()

#поднять/опустить хвост
up()
down()

# специально для создания разметки
goto(x,y) # переместиться на координаты x,y
dot(x) # поставить точку размером x

#ждет выхода из программы, не выходит сам
done()

```

 пример решения: 
 ![[Pasted image 20260217000421.png]]

```python
from turtle import *
#базавая штука
tracer(0)
screensize(2000, 2000)
lt(90)
kof = 30
#переместить чуть ниже
up()
backward(10*kof)
down()

# основная логика из задания
for i in range (2):
    forward(14*kof)
    lt(270)
    backward(12*kof)
    rt(90)
up()
forward(9*kof)
rt(90)
backward(7*kof)
lt(90)
down()
for i in range (2):
    forward(13*kof)
    rt(90)
    forward(6*kof)
    rt(90)
    
    #разметка для поля
up()
for x in range(-30, 30):
    for y in range(-30, 30):
        goto(x*kof, y*kof)
        dot(3)
        
        #ждет выхода
done()
```
